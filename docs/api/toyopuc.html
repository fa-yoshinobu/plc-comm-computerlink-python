160    def _recv_exact(self, n: int) -> bytes:
161        assert self._sock is not None
162        chunks = []
163        remaining = n
164        while remaining > 0:
165            try:
166                chunk = self._sock.recv(remaining)
167            except socket.timeout as e:
168                raise ToyopucTimeoutError('Receive timeout') from e
169            if not chunk:
170                raise ToyopucProtocolError('Connection closed while receiving')
171            chunks.append(chunk)
172            remaining -= len(chunk)
173        return b''.join(chunks)

175    def _send_and_recv(self, payload: bytes) -> ResponseFrame:
176        attempt = 0
177        last_err: Optional[Exception] = None
178        while attempt <= self.retries:
179            attempt += 1
180            if not self._sock:
181                self.connect()
182            assert self._sock is not None
183            self._last_tx = payload
184            self._last_rx = None
185
186            try:
187                if self.protocol == 'tcp':
188                    self._sock.sendall(payload)
189                    header = self._recv_exact(4)
190                    ll, lh = header[2], header[3]
191                    length = ll | (lh << 8)
192                    body = self._recv_exact(length)
193                    frame = header + body
194                else:
195                    self._sock.sendto(payload, (self.host, self.port))
196                    frame, _ = self._sock.recvfrom(self.recv_bufsize)
197            except socket.timeout as e:
198                last_err = ToyopucTimeoutError('Send/receive timeout')
199                if attempt <= self.retries:
200                    try:
201                        self.close()
202                    except Exception:
203                        pass
204                    import time
205
206                    time.sleep(self.retry_delay)
207                    continue
208                raise last_err from e
209            except OSError as e:
210                last_err = ToyopucError('Socket error')
211                if attempt <= self.retries:
212                    try:
213                        self.close()
214                    except Exception:
215                        pass
216                    import time
217
218                    time.sleep(self.retry_delay)
219                    continue
220                raise last_err from e
221
222            self._last_rx = frame
223            resp = parse_response(frame)
224            if resp.ft != FT_RESPONSE:
225                raise ToyopucProtocolError(f'Unexpected frame type: 0x{resp.ft:02X}')
226            if resp.rc != 0x00:
227                raise ToyopucError(format_response_error(resp))
228            return resp
